import numpy as np
import pandas as pd 
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans 

#made the values annual
#assuming an average person doesn't skydive more than 3 times a year 
#per 100,000 per year 


data = pd.DataFrame({
    "activity": ["Driving", "Skydiving", "Swimming", "Biking"],
    "fatality_rate": [12.9, 0.27*3, 1.5, 2.9]
})


#separate small numbers 
data["log_fatality"]= np.log(data["fatality_rate"])

X= data[["fatality_rate", "log_fatality"]].values

#run
kmeans= KMeans(n_clusters = 2, random_state= 42)
data["cluster"]=kmeans.fit_predict(X)


#label high or low risk
cluster_means = data.groupby("cluster")["fatality_rate"].mean()
high_risk = cluster_means.idxmax()
low_risk = cluster_means.idxmin()

data["risk_label"]= data["cluster"].apply(lambda x: "High Risk" if x==high_risk else "Low Risk")
colors = {"High Risk": "hotpink", "Low Risk": "blue"}

#plot
plt.figure(figsize=(5,5))
for risk in data["risk_label"].unique():
    #make the legend
    subset=data[data["risk_label"]== risk]
    plt.scatter(subset["fatality_rate"], subset["log_fatality"],
                s=200, label=risk, color = colors[risk])

for i, row in data.iterrows():
    plt.text(
        row["fatality_rate"] + 0.3,
        row["log_fatality"] + 0.02,
        row["activity"],
        fontsize=11,
        ha="left"
        )

plt.xlabel("Annual Fatality Rate per Population of 100,000")
plt.ylabel("Log of Annual Fatality Rate)")
plt.title("K-Means Clustering of Different Activities and their Fatality Risk")
plt.legend()
plt.tight_layout()
plt.show()
